```python
class PrintFormatProcessor:
    def __init__(self, print_format_name, doc_name):
        self.print_format_name = print_format_name
        self.doc_name = doc_name
        self.print_format_config = None
        self.template_engine = TemplateEngine()
        self.data_processor = PrintDataProcessor()
        self.cache_manager = PrintFormatCacheManager()
        self.performance_monitor = PrintFormatPerformanceMonitor()
        self.renderer = PrintRenderer()
        
        # Load print format configuration
        self.load_print_format_config()
    
    def load_print_format_config(self):
        """Load print format configuration from database"""
        self.print_format_config = frappe.get_doc('Print Format', self.print_format_name)
        
        if not self.print_format_config:
            raise PrintFormatException(f"Print Format {self.print_format_name} not found")
        
        # Parse print format configuration
        self.parse_print_format_config()
    
    def parse_print_format_config(self):
        """Parse print format configuration into internal format"""
        self.config = {
            'doctype': self.print_format_config.doc_type,
            'module': self.print_format_config.module,
            'template': self.print_format_config.html or self.build_default_template(),
            'css': self.print_format_config.css or self.get_default_css(),
            'font_size': self.print_format_config.font_size or 14,
            'page_size': self.print_format_config.page_size or 'A4',
            'orientation': self.print_format_config.orientation or 'Portrait',
            'margin_top': self.print_format_config.margin_top or 15,
            'margin_bottom': self.print_format_config.margin_bottom or 15,
            'margin_left': self.print_format_config.margin_left or 15,
            'margin_right': self.print_format_config.margin_right or 15,
            'line_height': self.print_format_config.line_height or 1.5,
            'format_type': self.print_format_config.format_type or 'Jinja',
            'raw_printing': self.print_format_config.raw_printing or 0
        }
    
    def build_default_template(self):
        """Build default template for print format"""
        doctype = self.config['doctype']
        
        # Get DocType meta
        meta = frappe.get_meta(doctype)
        
        # Build template with standard fields
        template = f"""
        <div class="print-format">
            <div class="print-header">
                <h2>{{{{ doc.name }}}}</h2>
                <div class="print-meta">
                    <p>Date: {{{{ doc.creation }}}}</p>
                    <p>Status: {{{{ doc.status }}}}</p>
                </div>
            </div>
            <div class="print-body">
        """
        
        # Add fields to template
        for field in meta.get('fields', []):
            if field.fieldtype not in ['Section Break', 'Column Break', 'Tab Break'] and field.in_list_view:
                template += f"""
                <div class="print-field">
                    <label>{field.label}:</label>
                    <span>{{{{ doc.{field.fieldname} or '' }}}}</span>
                </div>
                """
        
        template += """
            </div>
            <div class="print-footer">
                <p>Generated by Frappe</p>
            </div>
        </div>
        """
        
        return template

    def generate_print_format(self, filters=None, lang=None):
        """Generate print format with performance monitoring"""
        start_time = time.time()
        
        try:
            # Generate cache key
            cache_key = self.generate_cache_key(filters, lang)
            
            # Try to get from cache
            cached_result = self.cache_manager.get_cached_format(cache_key)
            if cached_result:
                return cached_result
            
            # Load document data
            doc_data = self.data_processor.load_document_data(
                self.config['doctype'], 
                self.doc_name, 
                filters
            )
            
            # Process template
            processed_template = self.template_engine.process_template(
                self.config['template'], 
                doc_data, 
                self.config['format_type']
            )
            
            # Apply CSS
            styled_template = self.apply_css(processed_template)
            
            # Generate final HTML
            final_html = self.renderer.render_html(styled_template, self.config)
            
            # Cache result
            self.cache_manager.cache_format(cache_key, final_html)
            
            # Track performance
            processing_time = time.time() - start_time
            self.performance_monitor.track_format_generation(
                self.print_format_name, 
                processing_time, 
                len(final_html)
            )
            
            return final_html
            
        except Exception as e:
            processing_time = time.time() - start_time
            self.performance_monitor.track_format_error(
                self.print_format_name, 
                processing_time, 
                str(e)
            )
            raise PrintFormatException(f"Print format generation failed: {str(e)}")
    
    def apply_css(self, html_content):
        """Apply CSS to HTML content"""
        # Create complete HTML document
        complete_html = f"""
        <!DOCTYPE html>
        <html>
        <head>
            <meta charset="utf-8">
            <title>{self.print_format_name}</title>
            <style>
                {self.config['css']}
            </style>
        </head>
        <body>
            {html_content}
        </body>
        </html>
        """
        
        return complete_html
    
    def generate_cache_key(self, filters, lang):
        """Generate cache key for print format"""
        cache_components = [
            self.print_format_name,
            self.doc_name,
            json.dumps(filters or {}, sort_keys=True),
            lang or frappe.session.lang,
            str(frappe.session.user)
        ]
        
        cache_string = '|'.join(cache_components)
        return f"print_format:{hashlib.md5(cache_string.encode()).hexdigest()}"
    
    def export_to_pdf(self, html_content, output_path=None):
        """Export print format to PDF"""
        try:
            import weasyprint
            
            # Configure WeasyPrint
            css = weasyprint.CSS(string=self.config['css'])
            
            # Create HTML document
            html_doc = weasyprint.HTML(string=html_content)
            
            # Generate PDF
            pdf_bytes = html_doc.write_pdf(stylesheets=[css])
            
            if output_path:
                with open(output_path, 'wb') as f:
                    f.write(pdf_bytes)
                return output_path
            else:
                return pdf_bytes
                
        except ImportError:
            # Fallback to wkhtmltopdf
            return self.export_to_pdf_wkhtmltopdf(html_content, output_path)
    
    def export_to_pdf_wkhtmltopdf(self, html_content, output_path=None):
        """Export to PDF using wkhtmltopdf"""
        import subprocess
        import tempfile
        import os
        
        # Create temporary HTML file
        with tempfile.NamedTemporaryFile(mode='w', suffix='.html', delete=False) as f:
            f.write(html_content)
            html_path = f.name
        
        try:
            # Generate PDF using wkhtmltopdf
            if output_path:
                cmd = [
                    'wkhtmltopdf',
                    '--page-size', self.config['page_size'],
                    '--orientation', self.config['orientation'].lower(),
                    '--margin-top', str(self.config['margin_top']),
                    '--margin-bottom', str(self.config['margin_bottom']),
                    '--margin-left', str(self.config['margin_left']),
                    '--margin-right', str(self.config['margin_right']),
                    html_path,
                    output_path
                ]
            else:
                # Generate to temporary file
                with tempfile.NamedTemporaryFile(suffix='.pdf', delete=False) as f:
                    output_path = f.name
                
                cmd = [
                    'wkhtmltopdf',
                    '--page-size', self.config['page_size'],
                    '--orientation', self.config['orientation'].lower(),
                    '--margin-top', str(self.config['margin_top']),
                    '--margin-bottom', str(self.config['margin_bottom']),
                    '--margin-left', str(self.config['margin_left']),
                    '--margin-right', str(self.config['margin_right']),
                    html_path,
                    output_path
                ]
            
            # Execute command
            result = subprocess.run(cmd, capture_output=True, text=True)
            
            if result.returncode != 0:
                raise PrintFormatException(f"PDF generation failed: {result.stderr}")
            
            # Read PDF content
            with open(output_path, 'rb') as f:
                pdf_content = f.read()
            
            return pdf_content
            
        finally:
            # Clean up temporary files
            try:
                os.unlink(html_path)
                if output_path and os.path.exists(output_path):
                    os.unlink(output_path)
            except:
                pass
```

```python
# Advanced template processing engine
class TemplateEngine:
    def __init__(self):
        self.jinja_env = self.setup_jinja_environment()
        self.template_cache = {}
        self.function_registry = self.register_template_functions()
    
    def setup_jinja_environment(self):
        """Setup Jinja2 environment with custom filters"""
        from jinja2 import Environment, FileSystemLoader
        
        env = Environment(
            loader=FileSystemLoader('.'),
            autoescape=True,
            trim_blocks=True,
            lstrip_blocks=True
        )
        
        # Register custom filters
        env.filters['currency'] = self.filter_currency
        env.filters['date'] = self.filter_date
        env.filters['datetime'] = self.filter_datetime
        env.filters['number'] = self.filter_number
        env.filters['percentage'] = self.filter_percentage
        env.filters['uppercase'] = self.filter_uppercase
        env.filters['lowercase'] = self.filter_lowercase
        env.filters['title'] = self.filter_title
        env.filters['truncate'] = self.filter_truncate
        env.filters['wordwrap'] = self.filter_wordwrap
        
        return env
    
    def register_template_functions(self):
        """Register custom template functions"""
        return {
            'get_doc': self.get_doc_function,
            'get_value': self.get_value_function,
            'get_link': self.get_link_function,
            'format_currency': self.format_currency_function,
            'format_date': self.format_date_function,
            'calculate_age': self.calculate_age_function,
            'get_image': self.get_image_function,
            'get_barcode': self.get_barcode_function,
            'get_qrcode': self.get_qrcode_function,
            'sum': self.sum_function,
            'avg': self.avg_function,
            'count': self.count_function,
            'min': self.min_function,
            'max': self.max_function
        }
    
    def process_template(self, template_string, data, format_type='Jinja'):
        """Process template with specified format type"""
        if format_type == 'Jinja':
            return self.process_jinja_template(template_string, data)
        elif format_type == 'HTML':
            return self.process_html_template(template_string, data)
        elif format_type == 'Raw':
            return template_string
        else:
            raise PrintFormatException(f"Unsupported template format: {format_type}")
    
    def process_jinja_template(self, template_string, data):
        """Process Jinja2 template"""
        try:
            # Create template
            template = self.jinja_env.from_string(template_string)
            
            # Add custom functions to template context
            context = data.copy()
            context.update(self.function_registry)
            
            # Render template
            rendered = template.render(**context)
            
            return rendered
            
        except Exception as e:
            raise PrintFormatException(f"Jinja template processing failed: {str(e)}")
    
    def process_html_template(self, template_string, data):
        """Process HTML template with simple variable replacement"""
        try:
            # Simple variable replacement
            rendered = template_string
            
            # Replace variables in format {{variable}}
            import re
            
            # Find all variables
            variables = re.findall(r'\{\{([^}]+)\}\}', rendered)
            
            for var in variables:
                # Clean variable name
                clean_var = var.strip()
                
                # Get value from data
                value = self.get_nested_value(data, clean_var)
                
                # Format value
                formatted_value = self.format_value(value)
                
                # Replace variable
                rendered = rendered.replace(f'{{{{{var}}}}}', formatted_value)
            
            return rendered
            
        except Exception as e:
            raise PrintFormatException(f"HTML template processing failed: {str(e)}")
    
    def get_nested_value(self, data, path):
        """Get nested value from data using dot notation"""
        keys = path.split('.')
        current = data
        
        for key in keys:
            if isinstance(current, dict) and key in current:
                current = current[key]
            elif hasattr(current, key):
                current = getattr(current, key)
            else:
                return None
        
        return current
    
    def format_value(self, value):
        """Format value for display"""
        if value is None:
            return ''
        elif isinstance(value, (int, float)):
            return str(value)
        elif isinstance(value, datetime.date):
            return value.strftime('%Y-%m-%d')
        elif isinstance(value, datetime.datetime):
            return value.strftime('%Y-%m-%d %H:%M:%S')
        else:
            return str(value)
    
    # Custom filters
    def filter_currency(self, value, currency=None):
        """Filter for currency formatting"""
        try:
            if currency is None:
                currency = frappe.db.get_default('currency')
            
            return frappe.format(value, {'fieldtype': 'Currency', 'options': currency})
        except:
            return str(value)
    
    def filter_date(self, value, format_string=None):
        """Filter for date formatting"""
        try:
            if format_string is None:
                format_string = frappe.db.get_single_value('System Settings', 'date_format') or 'yyyy-mm-dd'
            
            if isinstance(value, str):
                value = frappe.utils.getdate(value)
            
            return frappe.format(value, {'fieldtype': 'Date'})
        except:
            return str(value)
    
    def filter_datetime(self, value, format_string=None):
        """Filter for datetime formatting"""
        try:
            if isinstance(value, str):
                value = frappe.utils.get_datetime(value)
            
            return frappe.format(value, {'fieldtype': 'Datetime'})
        except:
            return str(value)
    
    def filter_number(self, value, decimals=None):
        """Filter for number formatting"""
        try:
            if decimals is not None:
                return f"{value:.{decimals}f}"
            else:
                return frappe.format(value, {'fieldtype': 'Float'})
        except:
            return str(value)
    
    def filter_percentage(self, value, decimals=1):
        """Filter for percentage formatting"""
        try:
            return f"{value:.{decimals}f}%"
        except:
            return str(value)
    
    def filter_uppercase(self, value):
        """Filter for uppercase conversion"""
        try:
            return str(value).upper()
        except:
            return str(value)
    
    def filter_lowercase(self, value):
        """Filter for lowercase conversion"""
        try:
            return str(value).lower()
        except:
            return str(value)
    
    def filter_title(self, value):
        """Filter for title case conversion"""
        try:
            return str(value).title()
        except:
            return str(value)
    
    def filter_truncate(self, value, length=50, suffix='...'):
        """Filter for text truncation"""
        try:
            text = str(value)
            if len(text) <= length:
                return text
            return text[:length] + suffix
        except:
            return str(value)
    
    def filter_wordwrap(self, value, width=80):
        """Filter for word wrapping"""
        try:
            import textwrap
            return textwrap.fill(str(value), width=width)
        except:
            return str(value)
    
    # Custom functions
    def get_doc_function(self, doctype, name):
        """Function to get document data"""
        try:
            doc = frappe.get_doc(doctype, name)
            return doc.as_dict()
        except:
            return {}
    
    def get_value_function(self, doctype, name, fieldname):
        """Function to get specific field value"""
        try:
            return frappe.db.get_value(doctype, name, fieldname)
        except:
            return None
    
    def get_link_function(self, doctype, name, label=None):
        """Function to generate link"""
        try:
            if label is None:
                label = name
            
            return f'<a href="/app/{doctype.lower().replace(' ', '-')}/{name}">{label}</a>'
        except:

    def format_currency_function(self, value, currency=None):
        """Function to format currency"""
        return self.filter_currency(value, currency)
    
    def format_date_function(self, value, format_string=None):
        """Function to format date"""

    def calculate_age_function(self, birth_date):
        """Function to calculate age from birth date"""
        try:
            if isinstance(birth_date, str):
                birth_date = frappe.utils.getdate(birth_date)
            
            today = frappe.utils.today()
            age = today.year - birth_date.year - ((today.month, today.day) < (birth_date.month, birth_date.day))
            
            return age
        except:
            return None

            if not image_name:
                return ''
            
            image_url = frappe.utils.get_url_to_file(image_name)

    def sum_function(self, values):
        """Function to calculate sum"""
        try:
            return sum(float(v) for v in values if v is not None)
        except:
            return 0
    
    def avg_function(self, values):
        """Function to calculate average"""
        try:
            valid_values = [float(v) for v in values if v is not None]
```

```python
# Advanced print data processing
class PrintDataProcessor:
    def __init__(self):
        self.field_formatters = self.register_field_formatters()
        self.child_table_processors = self.register_child_table_processors()
    
    def load_document_data(self, doctype, doc_name, filters=None):
        """Load and process document data for printing"""
        try:
            # Load main document
            doc = frappe.get_doc(doctype, doc_name)
            
            # Convert to dictionary
            data = doc.as_dict()
            
            # Apply field formatting
            data = self.apply_field_formatting(data, doctype)
            
            # Process child tables
            data = self.process_child_tables(data, doctype)
            
            # Add computed fields
            data = self.add_computed_fields(data, doctype)
            
            # Apply filters
            if filters:
                data = self.apply_filters(data, filters)
            
            return data
            
        except Exception as e:
            raise PrintFormatException(f"Failed to load document data: {str(e)}")
    
    def register_field_formatters(self):
        """Register field formatters"""
        return {
            'Currency': self.format_currency_field,
            'Date': self.format_date_field,
            'Datetime': self.format_datetime_field,
            'Time': self.format_time_field,
            'Percent': self.format_percent_field,
            'Int': self.format_int_field,
            'Float': self.format_float_field,
            'Link': self.format_link_field,
            'Select': self.format_select_field,
            'Check': self.format_check_field,
            'Text Editor': self.format_text_editor_field,
            'HTML': self.format_html_field,
            'Image': self.format_image_field,
            'Signature': self.format_signature_field
        }
    
    def register_child_table_processors(self):
        """Register child table processors"""
        return {
            'Sales Order Item': self.process_sales_order_item,
            'Purchase Order Item': self.process_purchase_order_item,
            'Journal Entry Account': self.process_journal_entry_account,
            'Payment Entry Reference': self.process_payment_entry_reference,
            'Timesheet Detail': self.process_timesheet_detail
        }
    
    def apply_field_formatting(self, data, doctype):
        """Apply field formatting to document data"""
        meta = frappe.get_meta(doctype)
        
        for field in meta.get('fields', []):
            fieldname = field.fieldname
            fieldtype = field.fieldtype
            
            if fieldname in data and fieldtype in self.field_formatters:
                formatter = self.field_formatters[fieldtype]
                data[fieldname] = formatter(data[fieldname], field)
        
        return data
    
    def process_child_tables(self, data, doctype):
        """Process child tables in document data"""
        meta = frappe.get_meta(doctype)
        child_table_fields = [f for f in meta.get('fields', []) if f.fieldtype == 'Table']
        
        for child_table_field in child_table_fields:
            options = child_table_field.options
            fieldname = child_table_field.fieldname
            
            if fieldname in data and data[fieldname]:
                # Process each child table row
                processed_rows = []
                
                for row in data[fieldname]:
                    processed_row = self.process_child_table_row(row, options)
                    processed_rows.append(processed_row)
                
                # Add computed fields for child table
                processed_rows = self.add_child_table_computed_fields(processed_rows, options)
                
                data[fieldname] = processed_rows
        
        return data
    
    def process_child_table_row(self, row, child_doctype):
        """Process individual child table row"""
        # Apply field formatting
        child_meta = frappe.get_meta(child_doctype)
        
        for field in child_meta.get('fields', []):
            fieldname = field.fieldname
            fieldtype = field.fieldtype
            
            if fieldname in row and fieldtype in self.field_formatters:
                formatter = self.field_formatters[fieldtype]
                row[fieldname] = formatter(row[fieldname], field)
        
        # Apply specific child table processor
        if child_doctype in self.child_table_processors:
            processor = self.child_table_processors[child_doctype]
            row = processor(row)
        
        return row
    
    def add_child_table_computed_fields(self, rows, child_doctype):
        """Add computed fields to child table"""
        for row in rows:
            # Add amount if not present
            if 'amount' not in row and 'qty' in row and 'rate' in row:
                try:
                    qty = float(row.get('qty', 0))
                    rate = float(row.get('rate', 0))
                    row['amount'] = qty * rate
                except:
                    row['amount'] = 0
            
            # Add formatted amount
            if 'amount' in row:
                row['amount_formatted'] = frappe.format(row['amount'], {'fieldtype': 'Currency'})
            
            # Add row number
            if 'idx' not in row:
                row['idx'] = rows.index(row) + 1
        
        return rows
    
    def add_computed_fields(self, data, doctype):
        """Add computed fields to document data"""
        # Add document URL
        data['doc_url'] = frappe.utils.get_url_to_form(doctype, data.get('name'))
        
        # Add print date
        data['print_date'] = frappe.utils.nowdate()
        data['print_time'] = frappe.utils.nowtime()
        data['print_datetime'] = frappe.utils.now()
        
        # Add company information
        company = frappe.db.get_value(doctype, data.get('name'), 'company')
        if company:
            data['company_name'] = frappe.db.get_value('Company', company, 'company_name')
            data['company_address'] = self.get_company_address(company)
            data['company_contact'] = self.get_company_contact(company)
        
        # Add user information
        data['printed_by'] = frappe.session.user
        data['printed_by_name'] = frappe.db.get_value('User', frappe.session.user, 'full_name')
        
        # Add totals for numeric fields
        data = self.calculate_totals(data, doctype)
        
        return data
    
    def calculate_totals(self, data, doctype):
        """Calculate totals for numeric fields"""
        meta = frappe.get_meta(doctype)
        
        # Find numeric fields
        numeric_fields = []
        for field in meta.get('fields', []):
            if field.fieldtype in ['Currency', 'Float', 'Int'] and field.fieldname in data:
                numeric_fields.append(field.fieldname)
        
        # Calculate totals
        for fieldname in numeric_fields:
            if isinstance(data[fieldname], (int, float)):
                data[f'{fieldname}_in_words'] = frappe.utils.in_words(data[fieldname])
        
        # Calculate child table totals
        child_table_fields = [f for f in meta.get('fields', []) if f.fieldtype == 'Table']
        
        for child_table_field in child_table_fields:
            fieldname = child_table_field.fieldname
            
            if fieldname in data and data[fieldname]:
                child_data = data[fieldname]
                
                # Calculate sum for amount fields
                if any('amount' in row for row in child_data):
                    total_amount = sum(float(row.get('amount', 0)) for row in child_data)
                    data[f'{fieldname}_total_amount'] = total_amount
                    data[f'{fieldname}_total_amount_formatted'] = frappe.format(total_amount, {'fieldtype': 'Currency'})
                    data[f'{fieldname}_total_amount_in_words'] = frappe.utils.in_words(total_amount)
                
                # Calculate sum for qty fields
                if any('qty' in row for row in child_data):
                    total_qty = sum(float(row.get('qty', 0)) for row in child_data)
                    data[f'{fieldname}_total_qty'] = total_qty
        
        return data
```

### 10.3 Performance Optimization for Print Formats

#### Caching and Optimization Strategies

```python
# Advanced caching and optimization for print formats
class PrintFormatCacheManager:
    def __init__(self):
        self.cache = frappe.cache()
        self.cache_ttl = {
            'template_cache': 3600,     # 1 hour
            'css_cache': 3600,          # 1 hour
            'document_cache': 300,     # 5 minutes
            'generated_cache': 1800    # 30 minutes
        }
        self.cache_stats = {
            'hits': 0,
            'misses': 0,
            'evictions': 0
        }
    
    def get_cached_template(self, template_name):
        """Get cached template"""
        cache_key = f"print_template:{template_name}"
        cached_template = self.cache.get(cache_key)
        
        if cached_template:
            self.cache_stats['hits'] += 1
            return cached_template
        
        self.cache_stats['misses'] += 1
        return None
    
    def cache_template(self, template_name, template_content):
        """Cache template content"""
        cache_key = f"print_template:{template_name}"
        ttl = self.cache_ttl['template_cache']
        self.cache.set(cache_key, template_content, expires_in_sec=ttl)
    
    def get_cached_css(self, css_name):
        """Get cached CSS"""
        cache_key = f"print_css:{css_name}"
        cached_css = self.cache.get(cache_key)
        
        if cached_css:
            self.cache_stats['hits'] += 1
            return cached_css
        
        self.cache_stats['misses'] += 1
        return None
    
    def cache_css(self, css_name, css_content):
        """Cache CSS content"""
        cache_key = f"print_css:{css_name}"
        ttl = self.cache_ttl['css_cache']
        self.cache.set(cache_key, css_content, expires_in_sec=ttl)
    
    def get_cached_document(self, doctype, doc_name):
        """Get cached document data"""
        cache_key = f"print_document:{doctype}:{doc_name}"
        cached_doc = self.cache.get(cache_key)
        
        if cached_doc:
            self.cache_stats['hits'] += 1
            return json.loads(cached_doc)
        
        self.cache_stats['misses'] += 1
        return None
    
    def cache_document(self, doctype, doc_name, doc_data):
        """Cache document data"""
        cache_key = f"print_document:{doctype}:{doc_name}"
        ttl = self.cache_ttl['document_cache']
        json_data = json.dumps(doc_data, default=str)
        self.cache.set(cache_key, json_data, expires_in_sec=ttl)
    
    def get_cached_format(self, format_key):
        """Get cached generated format"""
        cached_format = self.cache.get(format_key)
        
        if cached_format:
            self.cache_stats['hits'] += 1
            return cached_format
        
        self.cache_stats['misses'] += 1
        return None
    
    def cache_format(self, format_key, format_content):
        """Cache generated format"""
        ttl = self.cache_ttl['generated_cache']
        self.cache.set(format_key, format_content, expires_in_sec=ttl)
    
    def invalidate_document_cache(self, doctype, doc_name):
        """Invalidate document cache"""
        cache_key = f"print_document:{doctype}:{doc_name}"
        self.cache.delete(cache_key)
        self.cache_stats['evictions'] += 1
    
    def invalidate_template_cache(self, template_name):
        """Invalidate template cache"""
        cache_key = f"print_template:{template_name}"
        self.cache.delete(cache_key)
        self.cache_stats['evictions'] += 1
    
    def get_cache_stats(self):
        """Get cache statistics"""
        total_requests = self.cache_stats['hits'] + self.cache_stats['misses']
        hit_rate = self.cache_stats['hits'] / total_requests if total_requests > 0 else 0
        
        return {
            'hit_rate': hit_rate,
            'hits': self.cache_stats['hits'],
            'misses': self.cache_stats['misses'],
            'evictions': self.cache_stats['evictions']
        }

# Performance monitoring
class PrintFormatPerformanceMonitor:
    def __init__(self):
        self.performance_data = {}
        self.slow_formats = []
        self.error_log = []
    
    def track_format_generation(self, format_name, generation_time, output_size):
        """Track print format generation performance"""
        if format_name not in self.performance_data:
            self.performance_data[format_name] = {
                'generations': 0,
                'total_time': 0,
                'min_time': float('inf'),
                'max_time': 0,
                'avg_time': 0,
                'total_size': 0,
                'avg_size': 0
            }
        
        stats = self.performance_data[format_name]
        stats['generations'] += 1
        stats['total_time'] += generation_time
        stats['min_time'] = min(stats['min_time'], generation_time)
        stats['max_time'] = max(stats['max_time'], generation_time)
        stats['avg_time'] = stats['total_time'] / stats['generations']
        stats['total_size'] += output_size
        stats['avg_size'] = stats['total_size'] / stats['generations']
        
        # Track slow formats
        if generation_time > 3.0:  # 3 seconds threshold
            self.slow_formats.append({
                'format': format_name,
                'generation_time': generation_time,
                'output_size': output_size,
                'timestamp': frappe.utils.now()
            })
    
    def track_error(self, format_name, generation_time, error_message):
        """Track print format generation errors"""
        self.error_log.append({
            'format': format_name,
            'generation_time': generation_time,
            'error': error_message,
            'timestamp': frappe.utils.now()
        })
    
    def get_performance_report(self):
        """Get comprehensive performance report"""
        return {
            'formats': self.performance_data,
            'slow_formats': self.slow_formats,
            'errors': self.error_log,
            'summary': self.generate_summary()
        }
    
    def generate_summary(self):
        """Generate performance summary"""
        if not self.performance_data:
            return {}
        
        total_generations = sum(stats['generations'] for stats in self.performance_data.values())
        total_time = sum(stats['total_time'] for stats in self.performance_data.values())
        avg_time = total_time / total_generations if total_generations > 0 else 0
        
        slowest_format = max(
            self.performance_data.items(), 
            key=lambda x: x[1]['avg_time']
        ) if self.performance_data else None
        
        return {
            'total_generations': total_generations,
            'total_time': total_time,
            'avg_time': avg_time,
            'slowest_format': slowest_format[0] if slowest_format else None,
            'slowest_time': slowest_format[1]['avg_time'] if slowest_format else 0,
            'error_count': len(self.error_log),
            'slow_format_count': len(self.slow_formats)
        }
```

## Addendum: Server-Side PDF Generation & Default Print Format

### A.1 Server-Side PDF with `frappe.render_template()`

Use `frappe.render_template()` to render a Jinja2 template string or file path, then pass the result to `frappe.utils.pdf.get_pdf()` to produce a PDF binary.

```python
# your_app/utils/pdf_utils.py
import frappe
from frappe.utils.pdf import get_pdf

@frappe.whitelist()
def generate_asset_pdf(asset_name: str) -> dict:
    """
    Render an Asset document to PDF server-side and attach it to the record.
    Returns the file URL.
    """
    doc = frappe.get_doc("Asset", asset_name)

    # Render Jinja2 template — can be a template string or a path like
    # "asset_management/templates/asset_label.html"
    html = frappe.render_template(
        "asset_management/templates/asset_label.html",
        {"doc": doc},
    )

    # get_pdf() wraps wkhtmltopdf / WeasyPrint depending on site config
    pdf_bytes = get_pdf(html, options={"page-size": "A4", "orientation": "Portrait"})

    # Save as a private file attached to the document
    file_doc = frappe.get_doc(
        {
            "doctype": "File",
            "file_name": f"{asset_name}.pdf",
            "attached_to_doctype": "Asset",
            "attached_to_name": asset_name,
            "is_private": 1,
            "content": pdf_bytes,
        }
    )
    file_doc.save(ignore_permissions=True)

    return {"file_url": file_doc.file_url}
```

Key points:
- `frappe.render_template()` accepts either a template string or a path relative to any installed app's templates directory.
- `get_pdf()` respects the `pdf_engine` key in `site_config.json` (`wkhtmltopdf` by default, `weasyprint` if set).
- Always attach generated PDFs as `File` documents so they appear in the document's attachment list.

### A.2 Setting a Default Print Format via `hooks.py`

Register a default print format for a DocType so it is pre-selected whenever a user opens the print dialog.

```python
# your_app/hooks.py

# Map DocType → Print Format name (must exist in the database / fixtures)
default_print_format = {
    "Asset": "Asset Label",
    "Sales Invoice": "Custom Sales Invoice",
}
```

Frappe reads `default_print_format` from every installed app's `hooks.py` and merges them. The value must match the `name` field of an existing **Print Format** document. Export the Print Format as a fixture so it travels with the app:

```python
# hooks.py — also export the print format as a fixture
fixtures = [
    {"dt": "Print Format", "filters": [["doc_type", "in", ["Asset", "Sales Invoice"]]]},
]
```

Then run `bench export-fixtures` to write the JSON files into your app's `fixtures/` directory.
